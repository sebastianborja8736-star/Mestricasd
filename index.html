// VISUALIZACIÓN INTERACTIVA // MÉXICO vs ECUADOR con BANDERAS boolean mostrarMexico = false; boolean mostrarEcuador = false; float alturaMexico = 0; float alturaEcuador = 0; // Datos aproximados (2025) int valorMexico = 128000; int valorEcuador = 25000; int maxValor = 130000; void setup() { size(950, 550); smooth(); textAlign(CENTER); } void draw() { // Fondo violeta background(150, 120, 200); // Título fill(255); textSize(20); text("Personas Desaparecidas (datos acumulados hasta 2025)", width/2, 40); // ===== TEXTO CLICKEABLE ===== textSize(26); // MÉXICO fill(isHover(200, 90, 300, 130) ? color(200,230,255) : 255); text("MÉXICO", 250, 120); // ECUADOR fill(isHover(650, 90, 750, 130) ? color(200,230,255) : 255); text("ECUADOR", 700, 120); // ===== BANDERAS ===== drawFlagMexico(130, 150); drawFlagEcuador(800, 150); // ===== ANIMACIONES ===== float objetivoMexico = mostrarMexico ? map(valorMexico, 0, maxValor, 0, 260) : 0; float objetivoEcuador = mostrarEcuador ? map(valorEcuador, 0, maxValor, 0, 260) : 0; alturaMexico = lerp(alturaMexico, objetivoMexico, 0.08); alturaEcuador = lerp(alturaEcuador, objetivoEcuador, 0.08); // Color celeste fill(120, 200, 255); // Barras rect(250, height - 120 - alturaMexico, 45, alturaMexico); rect(700, height - 120 - alturaEcuador, 45, alturaEcuador); // Texto métricas fill(255); textSize(14); if (mostrarMexico) { text("128 000 personas", 270, height - 80); } if (mostrarEcuador) { text("25 000 personas", 720, height - 80); } // Línea base stroke(255, 120); line(100, height - 120, width - 100, height - 120); noStroke(); } void mousePressed() { // Click MÉXICO if (isHover(200, 90, 300, 130)) { mostrarMexico = !mostrarMexico; } // Click ECUADOR if (isHover(650, 90, 750, 130)) { mostrarEcuador = !mostrarEcuador; } } // ===== FUNCIONES ===== // Hover detector boolean isHover(int x1, int y1, int x2, int y2) { return mouseX > x1 && mouseX < x2 && mouseY > y1 && mouseY < y2; } // Bandera de México void drawFlagMexico(float x, float y) { float w = 60; float h = 40; noStroke(); fill(0, 140, 70); rect(x, y, w/3, h); fill(255); rect(x + w/3, y, w/3, h); fill(200, 30, 45); rect(x + 2*w/3, y, w/3, h); } // Bandera de Ecuador void drawFlagEcuador(float x, float y) { float w = 60; float h = 40; noStroke(); fill(255, 215, 0); // amarillo rect(x, y, w, h/2); fill(0, 56, 168); // azul rect(x, y + h/2, w, h/4); fill(206, 17, 38); // rojo rect(x, y + 3*h/4, w, h/4); }
